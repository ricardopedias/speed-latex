% Função para criar partes
\newcommand{\docPart}[1] {
    \part{
        \textcolor{part_text}{ #1 }
    }
}

% Função para criar o capítulo
\newcommand{\docChapter}[1] {
    \chapter{
        \textcolor{chapter_text}{ #1 }
    }
}
